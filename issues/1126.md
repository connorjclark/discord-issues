npc script customSounds

    void run(int spawnSound, int activeSound, int activeSoundMinInterval, int chance)
    {
        if(spawnSound) Audio->PlaySound(spawnSound);
        unless(activeSound) return;
        while(true)
        {
            Waitframes(activeSoundMinInterval);

    }
